## Cluster Configuration File ##

[cluster Slurm]
FormLayout = selectionpanel
Category = Schedulers

Autoscale = $Autoscale

    [[node defaults]]
    UsePublicNetwork = $UsePublicNetwork
    Credentials = $Credentials    
    SubnetId = $SubnetId
    Region = $Region
    KeyPairLocation = ~/.ssh/cyclecloud.pem
    
        [[[configuration]]]
        slurm.version = $configuration_slurm_version
        # Disable normal NFS exports and mounts
        cyclecloud.mounts.shared.disabled = ${NFSType != "Builtin"}
        cyclecloud.exports.shared.disabled = ${NFSType != "Builtin"}
        cyclecloud.exports.shared.samba.enabled = ${NFSType == "Builtin"}
        cyclecloud.mounts.sched.disabled = ${NFSType != "Builtin"}
        cyclecloud.exports.sched.disabled = ${NFSType != "Builtin"}
        cyclecloud.exports.sched.samba.enabled = ${NFSType == "Builtin"}
        cyclecloud.exports.defaults.samba.enabled = ${NFSType == "Builtin"}       
        cshared.server.legacy_links_disabled = ${NFSType != "Builtin"}

        [[[cluster-init cyclecloud/slurm:default]]]
        Optional = true

        [[[configuration cyclecloud.mounts.anf_shared]]]
        disabled = ${NFSType != "ANF"}
        type = nfs
        mountpoint = $ANFSharedMountpoint
        export_path = $ANFSharedExport
        address = $ANFSharedAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp

        [[[configuration cyclecloud.mounts.anf_sched]]]
        disabled = ${NFSType != "ANF"}
        type = nfs
        mountpoint = $ANFSchedMountpoint
        export_path = $ANFSchedExport
        address = $ANFSchedAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp

        [[[configuration cyclecloud.mounts.nfs_shared]]]
        disabled = ${NFSType != "NFS"}
        type = nfs
        export_path = $NFSSharedExportPath
        mountpoint = $NFSSharedMountPoint
        address = ${ifThenElse(NFSType == "NFS", regexps("^([^ ]+) .+$", NFSAddress, "\\1"), "")}

        [[[configuration cyclecloud.mounts.nfs_sched]]]
        disabled = ${NFSType != "NFS"}
        type = nfs
        export_path = $NFSSchedExportPath
        mountpoint = $NFSSchedMountPoint
        address = ${ifThenElse(NFSType == "NFS", regexps("^([^ ]+) .+$", NFSAddress, "\\1"), "")}

    [[node master]]
    MachineType = $MasterMachineType
    ImageName = $MasterImageName
    IsReturnProxy = $ReturnProxy
    AdditionalClusterInitSpecs = $MasterClusterInitSpecs
    
        [[[configuration]]]

        [[[cluster-init cyclecloud/slurm:master]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $UsePublicNetwork

        [[[input-endpoint ganglia]]]
        PrivatePort = 8652
        PublicPort = 8652


    [[nodearray hpc]]
    MachineType = $HPCMachineType
    ImageName = $HPCImageName
    MaxCoreCount = $MaxHPCExecuteCoreCount
    Azure.MaxScalesetSize = $HPCMaxScalesetSize
    AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs


        [[[configuration]]]
        slurm.autoscale = true
        slurm.default_partition = true
        slurm.hpc = true

        [[[cluster-init cyclecloud/slurm:execute]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic


    [[nodearray htc]]
    MachineType = $HTCMachineType
    ImageName = $HTCImageName
    MaxCoreCount = $MaxHTCExecuteCoreCount

    Interruptible = $HTCUseLowPrio
    AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs


        [[[configuration]]]
        slurm.autoscale = true
        slurm.hpc = false

        [[[cluster-init cyclecloud/slurm:execute]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic

[parameters Filer]
Order = 15

    [[parameter NFSType]]
    Label = NFS Type
    ParameterType = StringList
    Config.Label = Type of NFS to use for this cluster
    Config.Plugin = pico.form.Dropdown
    Config.Entries := {[Label="Azure NetApp Files (Medium to Large)"; Value="ANF"], [Label="NFS Cluster (Small to Medium)"; Value="NFS"], [Label="Use scheduler (Small)"; Value="Builtin"]}
    DefaultValue = Builtin
    #DefaultValue := {"ANF", "NFS", "Built-in"}

    [[parameter NFSAddress]]
    Label = NFS IP Address
    Description = The ip address of the NFS serve
    Config.Plugin = pico.form.QueryDropdown
    Config.Query = select StrJoin( " ", I.PrivateIp, "(", ClusterName, ")" ) as Value from Cloud.Node inner join Azure.Instance I on InstanceId===I.InstanceId where cluster().IsTemplate isnt true && Configuration.role=?="sharedfs"
    #"
    Config.ParameterType = String
    Config.PlaceHolderText = "Select or enter an IP address"
    Config.FreeForm = true
    Conditions.Excluded := NFSType != "NFS"

    [[parameter NFSSharedMountPoint]]
    Label = Shared Mount Point
    Description = The path at which to mount the Filesystem
    DefaultValue = /shared
    Conditions.Excluded := NFSType != "NFS"

    [[parameter NFSSharedExportPath]]
    Label = Shared Export Path
    Description = The path exported by the file system
    DefaultValue = /mnt/exports/data/shared
    Conditions.Excluded := NFSType != "NFS"

    [[parameter NFSSchedMountPoint]]
    Label = Sched Mount Point
    Description = The path at which to mount the Filesystem
    DefaultValue = /sched
    Conditions.Excluded := NFSType != "NFS"

    [[parameter NFSSchedExportPath]]
    Label = Sched Export Path
    Description = The path exported by the file system
    DefaultValue = /mnt/exports/data/sched
    Conditions.Excluded := NFSType != "NFS"

    [[parameter ANFSharedMountpoint]]
    Label = ANF Shared Mountpoint
    Config.Label = Path to mount ANF shared volume
    DefaultValue = /shared
    Conditions.Excluded := NFSType != "ANF"

    [[parameter ANFSharedExport]]
    Label = ANF Shared Export
    Config.Label = ANF Shared export to mount
    DefaultValue = /shared
    Conditions.Excluded := NFSType != "ANF"

    [[parameter ANFSharedAddress]]
    Label = ANF Shared Filer Address
    Config.Label = ANF IP or hostname
    DefaultValue = localhost
    Conditions.Excluded := NFSType != "ANF"

    [[parameter ANFSchedMountpoint]]
    Label = ANF Scheduler Mountpoint
    Config.Label = Path to mount ANF scheduler volume
    DefaultValue = /sched
    Conditions.Excluded := NFSType != "ANF"

    [[parameter ANFSchedExport]]

    [[parameter ANFSchedAddress]]

    Config.Label = ANF IP or hostname
