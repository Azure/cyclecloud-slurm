################################
## Cluster Configuration File ##

[cluster Slurm]
FormLayout = selectionpanel
Category = Schedulers

Autoscale = $Autoscale

    [[node defaults]]
    UsePublicNetwork = $UsePublicNetwork
    Credentials = $Credentials    
    SubnetId = $SubnetId
    Region = $Region
    KeyPairLocation = ~/.ssh/cyclecloud.pem
    
        [[[configuration]]]
        slurm.version = $configuration_slurm_version
        # Disable normal NFS exports and mounts
        cyclecloud.mounts.shared.disabled = ${NFSType != "Builtin"}
        cyclecloud.exports.shared.disabled = ${NFSType != "Builtin"}

        cyclecloud.mounts.sched.disabled = false
        cyclecloud.exports.sched.disabled = false
        cyclecloud.exports.sched.samba.enabled = false
        cyclecloud.exports.shared.samba.enabled = false
        cyclecloud.exports.defaults.samba.enabled = false      

        cshared.server.legacy_links_disabled = ${NFSType != "Builtin"}

        [[[cluster-init cyclecloud/slurm:default]]]
        Optional = true

        [[[configuration cyclecloud.mounts.nfs_shared]]]
        disabled = ${NFSType != "External"}
        type = nfs
        export_path = $NFSSharedExportPath
        mountpoint = $NFSSharedMountPoint
        address = ${ifThenElse(NFSType == "External", regexps("^([^ ]+) .+$", NFSAddress, "\\1"), "")}
        options = $NFSSharedMountOptions

        [[[configuration cyclecloud.mounts.additional_nfs]]]
        disabled = ${AdditionalNAS isnt true}
        type = nfs
        address = $AdditonalNFSAddress
        mountpoint = $AdditionalNFSMountPoint
        export_path = $AdditionalNFSExportPath
        options = $AdditionalNFSMountOptions

    [[node master]]
    MachineType = $MasterMachineType
    ImageName = $MasterImageName
    IsReturnProxy = $ReturnProxy
    AdditionalClusterInitSpecs = $MasterClusterInitSpecs
    
        [[[configuration]]]

        [[[cluster-init cyclecloud/slurm:master]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $UsePublicNetwork

        [[[volume sched]]]
        Size = 1024
        SSD = True
        Mount = sched
        Persistent = False

        [[[configuration cyclecloud.mounts.nfs_sched]]]
        mountpoint = /mnt/exports/sched
        fs_type = xfs

        [[[configuration cyclecloud.exports.nfs_sched]]]
        export_path = /mnt/exports/sched
        options = no_root_squash
        samba.enabled = false
        type = nfs

        [[[volume nfs-shared]]]
        Size = $FilesystemSize
        SSD = True
        Mount = shared
        Persistent = $FileSystemPersistence

        [[[configuration cyclecloud.mounts.shared]]]
        disabled = ${NFSType != "Builtin"}
        mountpoint = /shared
        fs_type = xfs

        [[[configuration cyclecloud.exports.shared]]]
        disabled = ${NFSType != "Builtin"}
        export_path = /shared
        samba.enabled = false
        type = nfs


    [[nodearray hpc]]
    MachineType = $HPCMachineType
    ImageName = $HPCImageName
    MaxCoreCount = $MaxHPCExecuteCoreCount
    Azure.MaxScalesetSize = $HPCMaxScalesetSize
    AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs


        [[[configuration]]]
        slurm.autoscale = true
        slurm.default_partition = true
        slurm.hpc = true

        [[[cluster-init cyclecloud/slurm:execute]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic


    [[nodearray htc]]
    MachineType = $HTCMachineType
    ImageName = $HTCImageName
    MaxCoreCount = $MaxHTCExecuteCoreCount

    Interruptible = $HTCUseLowPrio
    AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs


        [[[configuration]]]
        slurm.autoscale = true
        slurm.hpc = false

        [[[cluster-init cyclecloud/slurm:execute]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic

        [[[parameter slurm]]]

[parameters Network Attached Storage]
Order = 15

    [[parameters /shared]]
        [[[parameter About shared]]]
        HideLabel = true
        Config.Plugin = pico.widget.HtmlTemplateWidget
        Config.Template := "<p>The directory <code>/shared</code> is a network attached mount and exists in all nodes of the cluster.<br><br>There are two options for providing this mount:<br> <strong>[Builtin]</strong>: The scheduler node is an NFS server that provides the mountpoint to the other nodes of the cluster.<br> <strong>[External]</strong>: A network attached storage such as Azure Netapp Files, HPC Cache, or another VM running an NFS server, provides the mountpoint.</p>"
        Order = 20

    [[[parameter NFSType]]]
    Label = NFS Type
    ParameterType = StringList
    Config.Label = Type of NFS to use for this cluster
    Config.Plugin = pico.form.Dropdown
    Config.Entries := {[Label="External NFS"; Value="External"], [Label="Use Builtin"; Value="Builtin"]}
    DefaultValue = Builtin

    [[[parameter NFSAddress]]]
    Label = NFS IP Address
    Description = The ip address of the NFS serve
    Config.Plugin = pico.form.QueryDropdown
    Config.Query = select StrJoin( " ", I.PrivateIp, "(", ClusterName, ")" ) as Value from Cloud.Node inner join Azure.Instance I on InstanceId===I.InstanceId where cluster().IsTemplate isnt true && Configuration.role=?="sharedfs"
    #"
    Config.ParameterType = String
    Config.PlaceHolderText = "Select or enter an IP address"
    Config.FreeForm = true
    Conditions.Excluded := NFSType != "External"

    [[[parameter NFSSharedMountPoint]]]
    Label = Shared Mount Point
    Description = The path at which to mount the Filesystem
    DefaultValue = /shared
    Conditions.Excluded := NFSType != "External"

    [[[parameter NFSSharedExportPath]]]
    Label = Shared Export Path
    Description = The path exported by the file system
    DefaultValue = /mnt/exports/shared
    Conditions.Excluded := NFSType != "External"

    [[[parameter NFSSharedMountOptions]]]
    Label = NFS Mount Options
    Description = NFS Client Mount Options
    Conditions.Excluded := NFSType != "External"

    [[[parameter FilesystemSize]]]
    Label = Size (GB)
    Description = The filesystem size
    DefaultValue = 100

    Config.Plugin = pico.form.NumberTextBox
    Config.MinValue = 10
    Config.MaxValue = 10240
    Config.IntegerOnly = true
    Conditions.Excluded := NFSType != "Builtin"

    [[[parameter FileSystemPersistence]]]
    Conditions.Excluded := NFSType != "Builtin"
    Label = Filesystem Persistence
    DefaultValue = true
    Config.Plugin = pico.form.RadioButtonsControl
        [[[[list Config.Entries]]]]
        Name = false
        Label = Ephemeral
        [[[[list Config.Entries]]]]
        Name = true
        Label = Persistent

    [[parameters Additional NFS Mount]]
        [[[parameter Additional NFS Mount Readme]]]
        HideLabel = true
        Config.Plugin = pico.widget.HtmlTemplateWidget
        Config.Template := "<p>Mount another NFS endpoint on the cluster nodes</p>"
        Order = 20

        [[[parameter AdditionalNAS]]]
        HideLabel = true
        DefaultValue = false
        Widget.Plugin = pico.form.BooleanCheckBox
        Widget.Label = Add NFS mount

    [[[parameter AdditonalNFSAddress]]]
    Label = NFS IP Address or hostname
    Description = The ip address or hostname of the NFS server
    Config.ParameterType = String
    Conditions.Excluded := AdditionalNAS isnt true

    [[[parameter AdditionalNFSMountPoint]]]

    [[[parameter AdditionalNFSExportPath]]]

    Conditions.Excluded := AdditionalNAS isnt true

    [[[parameter AdditionalNFSMountOptions]]]

    Conditions.Excluded := AdditionalNAS isnt true
